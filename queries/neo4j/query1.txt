MATCH (r:Rental)
WITH r
ORDER BY r.rental_date DESC LIMIT 1
with datetime({epochmillis:r.rental_date}).year as most_recent_year

match (cat2:Category)<-[:IN_CATEGORY]-(f2:Film)<-[:RENTS_FILM]-(:Inventory)<-[:RENTS]-(r2:Rental)-[:RENTED_TO]->
(customer:Customer)<-[:RENTED_TO]-(r1:Rental)-[:RENTS]->(:Inventory)-[:RENTS_FILM]->(f1:Film)-[:IN_CATEGORY]->(cat1:Category)

where cat2 <> cat1 and f1 <> f2 and
datetime({epochmillis:r1.rental_date}).year = most_recent_year and datetime({epochmillis:r2.rental_date}).year = most_recent_year
return DISTINCT customer.customer_id
